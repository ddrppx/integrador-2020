abstract class Produto {
	atributos {
		nome
		valor
		desconto
		promocao
	}

	metodos {
		getNome()
		setNome(nome)
		getValor()
		setValor(valor)
		getDesconto()
		setDesconto(desconto)
		__valorTotal()
		__construct(nome, valor)
		addPromocao(Promocao promocao)
		remPromocao()
		__toString()
	}
}
